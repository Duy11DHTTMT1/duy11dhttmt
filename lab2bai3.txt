    public static void main(String... args) {
        long soKyDien, tien = 0;
        Scanner scanner = new Scanner(System.in);
        System.out.print("Nhap vào so ký dien cua tháng = ");
        soKyDien = scanner.nextLong();
        System.out.println("Chuong trình tính tien dien");

        do {
            System.out.print("Nhap vào so ký dien cua tháng = ");
            soKyDien = scanner.nextLong();
            if (soKyDien <= 0)
                System.out.println("So ký dien không duoc nho hon hoac bang 0");
        } while (soKyDien <= 0);

        if (soKyDien < 50) {
            tien = soKyDien * 1000;
        } else {
            tien = 50 * 1000 + (soKyDien - 50) * 1200;
        }
        System.out.println("Tien dien phai tra là: " + tien);
        scanner.close();
    }
}
